🚀 QUICK START INSTRUCTIONS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1️⃣  INSTALL DEPENDENCIES (1 minute)
      $ pip install -r requirements.txt

  2️⃣  GET API KEY (5 minutes)
      • Go to https://console.cloud.google.com/
      • Create project & enable YouTube Data API v3
      • Generate API key
      • Add to .env: YOUTUBE_API_KEY=your_key

  3️⃣  SETUP DATABASE (5 minutes)
      $ createdb youtube_db
      • Update config.py with DB credentials

  4️⃣  RUN APPLICATION (1 minute)
      $ streamlit run app.py
      • Opens at http://localhost:8501

  5️⃣  START USING! 
      • Go to "Fetch Data" tab
      • Enter a channel ID
      • View analytics in Dashboard

💾 DATABASE STRUCTURE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  PostgreSQL (youtube_db)
  │
  ├─ channels table
  │  ├─ channel_id (PK)
  │  ├─ title, description
  │  ├─ subscribers, total_views, total_videos
  │  ├─ custom_url, published_at
  │  └─ profile_image, banner_image, fetched_at
  │
  └─ videos table
     ├─ video_id (PK)
     ├─ channel_id (FK) → channels
     ├─ title, description, duration
     ├─ views, likes, comments
     ├─ published_at, thumbnail
     └─ fetched_at
